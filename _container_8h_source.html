109  bool PropagateMouseEvent(P param)
110  {
111  if (children.empty()) return false;
112 
113  // We iterate over the child widgets in reverse since the widgets
114  // are rendered back to front, so we let the ones in front get
115  // first chance to handle the envents.
116 
117  Vec2 oldOrigin(0, 0);
118  BOOST_REVERSE_FOREACH(auto &child, children) {
119  oldOrigin = display.AddUiOrigin(child->GetPos());
120  bool retv = (child.get()->*F)(param);
121  display.SetUiOrigin(oldOrigin);
122  if (retv) return true;
123  }
124 
125  return false;
126  }
