161  ReorderChild(const std::shared_ptr<T> &child, size_t idx)
162  {
163  auto sz = children.size();
164  if (sz == 0) return child;
165 
166  if (idx >= sz) {
167  idx = sz - 1;
168  }
169 
170  for (auto iter = children.begin(); iter != children.end(); ++iter) {
171  if (iter->child == child) {
172  auto dest = children.begin();
173  std::advance(dest, idx);
174  if (dest < iter) {
175  std::rotate(dest, iter, iter + 1);
176  }
177  else if (dest > iter) {
178  std::rotate(iter, iter + 1, dest + 1);
179  }
180  break;
181  }
182  }

196  template<typename P, bool(UiViewModel::*F)(P)>
197  bool PropagateMouseEvent(P param)
198  {
199  if (!visible || children.empty()) return false;
200 
201  // We iterate over the child widgets in reverse since the widgets
202  // are rendered back to front, so we let the ones in front get
203  // first chance to handle the events.
204 
205  Vec2 oldOrigin(0, 0);
206  for (auto iter = children.rbegin();
207  iter != children.rend(); ++iter)
208  {
209  auto &child = iter->child;
210  oldOrigin = display.AddUiOrigin(child->GetPos());
211  bool retv = (child.get()->*F)(param);
212  display.SetUiOrigin(oldOrigin);
213  if (retv) return true;
214  }
215 
216  return false;
217  }
