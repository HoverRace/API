143  template<typename P, bool(UiViewModel::*F)(P)>
144  bool PropagateMouseEvent(P param)
145  {
146  if (!visible || children.empty()) return false;
147 
148  // We iterate over the child widgets in reverse since the widgets
149  // are rendered back to front, so we let the ones in front get
150  // first chance to handle the events.
151 
152  Vec2 oldOrigin(0, 0);
153  for (auto iter = children.rbegin();
154  iter != children.rend(); ++iter)
155  {
156  auto &child = *iter;
157  oldOrigin = display.AddUiOrigin(child->GetPos());
158  bool retv = (child.get()->*F)(param);
159  display.SetUiOrigin(oldOrigin);
160  if (retv) return true;
161  }
162 
163  return false;
164  }
