160  void SetValue(const T &val)
161  {
162  // Find the first item matching this value.
163  // We assume that the list is small enough that we don't need a
164  // lookup table.
165  DefaultItem *newSel = nullptr;
166  for (const auto &item : items) {
167  if (item->GetValue() == val) {
168  newSel = item.get();
169  break;
170  }
171  }
172 
173  if (selItem != newSel) {
174  if (selItem) {
175  selItem->SetChecked(false);
176  selItem = newSel;
177  }
178  if (newSel) {
