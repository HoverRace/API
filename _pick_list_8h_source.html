185  void SetValue(const T &val)
186  {
187  // Find the first item matching this value.
188  // We assume that the list is small enough that we don't need a
189  // lookup table.
190  boost::optional<size_t> newSel;
191  for (auto idx : filteredItems) {
192  if (items[idx].item.GetValue() == val) {
193  newSel = boost::make_optional(idx);
194  break;
195  }
196  }
197 
198  if (selItem != newSel) {
199  if (selItem) {
200  items[*selItem].item.SetChecked(false);
201  selItem = newSel;
202  }
203  if (newSel) {

205  }
206  }

238  template<class Fn>
239  void ApplyFilter(Fn fn)
240  {
241  filteredItems.clear();
