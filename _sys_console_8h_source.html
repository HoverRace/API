134  template<class Function>
135  void ReadLogs(int start, int end, Function fn)
136  {
137  if (logLines.empty()) return;
138  if (start > end) return;
139  int endIdx = baseLogIdx + logLines.size() - 1;
140  if (start > endIdx) return;
141  if (start == end) fn(logLines.at(start - baseLogIdx));
142  else {
143  if (start < baseLogIdx) start = baseLogIdx;
144  auto startIter = logLines.begin() + (start - baseLogIdx);
145  auto endIter = (end > endIdx) ? logLines.end() : startIter + (end - start + 1);
146  std::for_each(startIter, endIter, fn);
147  }
148  }
