146  template<class Function>
147  void ReadLogs(int start, int end, Function fn)
148  {
149  if (logLines.empty()) return;
150  if (start > end) return;
151  int endIdx = baseLogIdx + logLines.size() - 1;
152  if (start > endIdx) return;
153  if (start == end) fn(logLines.at(start - baseLogIdx));
154  else {
155  if (start < baseLogIdx) start = baseLogIdx;
156  auto startIter = logLines.begin() + (start - baseLogIdx);
157  auto endIter = (end > endIdx) ? logLines.end() : startIter + (end - start + 1);
158  std::for_each(startIter, endIter, fn);
159  }
160  }
