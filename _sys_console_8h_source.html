145  template<class Function>
146  void ReadLogs(int start, int end, Function fn)
147  {
148  if (logLines.empty()) return;
149  if (start > end) return;
150  int endIdx = baseLogIdx + logLines.size() - 1;
151  if (start > endIdx) return;
152  if (start == end) fn(logLines.at(start - baseLogIdx));
153  else {
154  if (start < baseLogIdx) start = baseLogIdx;
155  auto startIter = logLines.begin() + (start - baseLogIdx);
156  auto endIter = (end > endIdx) ? logLines.end() : startIter + (end - start + 1);
157  std::for_each(startIter, endIter, fn);
158  }
159  }
