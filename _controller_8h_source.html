235  // Hashing scheme (we have 32 bits but won't always use them):
236  // disabled control
237  // [000000000000000000][aaaaaaaaaaaa]
238  // a: next available disabled id
239  // keyboard event
240  // [00000000][00][000000][aaaaaaaa][00000000]: keycode
241  // a: int keycode
242  // [00000000][00][01][aaaaaaaaaaaa][00000000]: scancode as keycode
243  // a: int scancode as keycode
244  // mouse event
245  // [00000000][01][00][aaaaaaaa][000000000000]: button press
246  // a: button id
247  // [00000000][01][01][aaaa][bbbb][000000000000]: axis move
248  // a: axis id
249  // b: direction
250  // joystick event
251  // [00000000][10][00][aaaaaaaa][bbbbbbbb][0000]: button press
252  // a: joystick id
253  // b: button id
254  // [00000000][10][01][aaaaaaaa][bbbbbbbb][0000]: slider move
255  // a: joystick id
256  // b: slider id
257  // [00000000][10][10][aaaaaaaa][bbbb][cccc][0000]: pov move
258  // a: joystick id
