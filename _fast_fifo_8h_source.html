28 template <class pType>
29 class MR_FastFifoBase
30 {
31 
32  protected:
33  int mHead;
34  int mNbItem;
35  int mArraySize;
36  pType *mArray;
37 
38  public:
39 
40  MR_FastFifoBase<pType>(int pSize, pType * pData) {
41  mNbItem = 0;
42  mHead = 0;
43  mArraySize = pSize;
44  mArray = pData;
45  };
46 
47  void Add(pType pData) {
48  mArray[(mHead + mNbItem) % mArraySize] = pData;
49  if(mNbItem == mArraySize) {
50  mHead = (mHead + 1) % mArraySize;
51  }
52  else {
53  mNbItem++;
54  }
55  };
56  void Remove(int pCount = 1) {
57  if(pCount > mNbItem) {
58  pCount = mNbItem;
59  }
60  mNbItem -= pCount;
61  mHead = (mHead + pCount) % mArraySize;
62  };
63  pType GetHead() {
64  return mArray[mHead];
65  };
66  void Clean() {
67  mNbItem = 0;
68  }
69  int Used() const
70  {
71  return mNbItem;
72  }
73  int TotalSize() const
74  {
75  return mArraySize;
76  }
77  int Full() const
78  {
79  return mNbItem == mArraySize;
80  }
81  BOOL CanAdd(int pCount = 1) const
82  {
83  return mNbItem + pCount <= mArraySize;
84  }
85  BOOL IsEmpty() const
86  {
87  return (mNbItem == 0);
88  }
89  const pType operator[] (int pIndex) const
90  {
91  return mArray[(mHead + pIndex) % mArraySize];
92  };
93 
94  // Special functions to add at the head and remove at the tail
95  void AddHead(pType pData) {
96  mHead = (mHead + mArraySize - 1) % mArraySize;
97  mArray[mHead] = pData;
98  if(mNbItem != mArraySize) {
99  mNbItem++;
100  }
101  };
102  void RemoveTail(int pCount = 1) {
103  if(pCount > mNbItem) {
104  pCount = mNbItem;
105  }
106  mNbItem -= pCount;
107  };
108 
109 };
